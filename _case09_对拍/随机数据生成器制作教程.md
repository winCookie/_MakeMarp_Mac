# 随机数据生成器制作教程

## 1. 基础随机数生成

### 1.1 最简单的随机数
```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    srand(time(0)); // 设置随机种子
    
    int n = rand() % 100 + 1; // 生成1~100的随机数
    cout << n << endl;
    
    return 0;
}
```

### 1.2 更高质量的随机数（推荐）
```cpp
#include <bits/stdc++.h>
#include <random>
using namespace std;

int main() {
    mt19937 rng(time(0)); // 更好的随机数生成器
    
    int n = uniform_int_distribution<int>(1, 100)(rng);
    cout << n << endl;
    
    return 0;
}
```

## 2. 常见数据类型的生成

### 2.1 生成整数数组
```cpp
// 生成长度为n，数值在[L,R]范围内的数组
void generate_array(int n, int L, int R) {
    mt19937 rng(time(0));
    
    cout << n << endl;
    for (int i = 0; i < n; i++) {
        int x = uniform_int_distribution<int>(L, R)(rng);
        cout << x << " ";
    }
    cout << endl;
}
```

### 2.2 生成字符串
```cpp
// 生成长度为n的随机字符串
string generate_string(int n, string charset = "abcdefghijklmnopqrstuvwxyz") {
    mt19937 rng(time(0));
    string s;
    
    for (int i = 0; i < n; i++) {
        int idx = uniform_int_distribution<int>(0, charset.size()-1)(rng);
        s += charset[idx];
    }
    return s;
}
```

### 2.3 生成树结构
```cpp
// 生成n个节点的树
void generate_tree(int n) {
    mt19937 rng(time(0));
    
    cout << n << endl;
    for (int i = 2; i <= n; i++) {
        int father = uniform_int_distribution<int>(1, i-1)(rng);
        cout << father << " " << i << endl;
    }
}
```

## 3. 实用技巧

### 3.1 控制数据范围
```cpp
const int MAXN = 1000;     // 最大数据规模
const int MIN_VAL = 1;     // 最小值
const int MAX_VAL = 10000; // 最大值

int main() {
    mt19937 rng(time(0));
    
    // 随机数据规模
    int n = uniform_int_distribution<int>(1, MAXN)(rng);
    
    // 随机数值
    int x = uniform_int_distribution<int>(MIN_VAL, MAX_VAL)(rng);
}
```

### 3.2 生成特殊数据

#### 有序数组
```cpp
// 生成有序数组（可控制升降序）
void generate_sorted_array(int n, bool ascending = true) {
    vector<int> arr(n);
    iota(arr.begin(), arr.end(), 1); // 填充1,2,3,...,n
    
    if (!ascending) {
        reverse(arr.begin(), arr.end());
    }
    
    cout << n << endl;
    for (int x : arr) cout << x << " ";
    cout << endl;
}
```

#### 生成图
```cpp
// 生成n个点m条边的图
void generate_graph(int n, int m) {
    mt19937 rng(time(0));
    
    cout << n << " " << m << endl;
    for (int i = 0; i < m; i++) {
        int u = uniform_int_distribution<int>(1, n)(rng);
        int v = uniform_int_distribution<int>(1, n)(rng);
        cout << u << " " << v << endl;
    }
}
```

## 4. 完整示例模板

```cpp
#include <bits/stdc++.h>
using namespace std;

// 全局随机数生成器
mt19937 rng(time(0));

// 生成[l, r]范围内的随机整数
int rand_int(int l, int r) {
    return uniform_int_distribution<int>(l, r)(rng);
}

// 生成随机数组
vector<int> generate_array(int n, int l, int r) {
    vector<int> arr(n);
    for (int i = 0; i < n; i++) {
        arr[i] = rand_int(l, r);
    }
    return arr;
}

int main() {
    // 数据规模
    int n = rand_int(1, 100);
    
    // 生成数组
    vector<int> arr = generate_array(n, 1, 1000);
    
    // 输出
    cout << n << endl;
    for (int x : arr) cout << x << " ";
    cout << endl;
    
    return 0;
}
```

## 5. 注意事项

1. **设置随机种子**：使用 `time(0)` 确保每次运行结果不同
2. **控制数据范围**：根据题目要求设置合理的上下界
3. **避免重复**：对拍时需要生成不同的测试数据
4. **边界情况**：记得测试最小/最大规模的数据
